def call(tag)
  case tag
  when Tag
    if tag.type.to_s == 'input' && %w'date datetime datetime-local'.include?((tag.attr[:type] || tag.attr['type']).to_s)
      attr = tag.attr.dup
      attr.delete(:type)
      attr.delete('type')
      attr['type'] = 'text'
      "<#{tag.type}#{attr_html(attr)}/>"
    else
      super
    end
  else
    super
  end
end
