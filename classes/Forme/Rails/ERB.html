def forme(obj=nil, attr={}, opts={}, &block)
  h = {:template=>self, :hidden_tags=>Forme::Rails::HIDDEN_TAGS}
  (obj.is_a?(Hash) ? attr = attr.merge(h) : opts = opts.merge(h))
  Form.form(obj, attr, opts, &block)
end
